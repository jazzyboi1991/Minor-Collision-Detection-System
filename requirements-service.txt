# requirements-service.txt
# 물피도주 자동감지 "웹 서비스" 전체 스택 (웹 + Celery 워커 + ML 추론)
# 재현 검증 환경: Python 3.11.15 (Linux/WSL2, CPU 추론) 
#
# ⚠️ 이 파일은 서비스 실행용(CPU)입니다. GPU 학습 환경은 AI 개발 브랜치 requirements.txt 참고.
# ⚠️ Python 3.12+/3.14 는 torch 휠 미제공 → 반드시 Python 3.11 환경에서 설치.
# ================================================================
#
# ── 웹 서비스 실행에 "필요한" 프로젝트 구성 ───────────────────────
#   아래 항목을 구성하여 서비스 실행. (그 외 data/ 문서/ 학습용 등은 실행 불필요)
#
#   capstone-26/
#   ├── backend/               [필수] FastAPI 웹 + Celery 워커 (서비스 본체)
#   ├── model/                 [필수] 추론 코드 (워커가 import: config, hitandrun_model,
#   │                                 predict_cam, device_utils, __init__)
#   ├── weights/               [필수] 학습된 가중치 hitandrun_model_best.pth (없으면 추론 불가)
#   ├── ui/                    [필수] 프론트엔드 소스 (src/, package.json, vite.config.js, index.html)
#   ├── docker-compose.yml     [필수] MySQL8 + Redis 기동 (또는 외부로 동등 서비스 제공)
#   ├── requirements-service.txt  [필수] 본 파일 (Python 3.11 환경 구성용)
#   │
#   ├── storage/               [자동생성] 업로드 원본/CAM 클립 — settings.py가 자동 mkdir
#   ├── venv311/               [각자설치생성] 본 파일로 만드는 Python 3.11 환경
#   └── ui/node_modules/       [각자설치생성] `npm install` 로 생성
#
#   ※ 실행 전제: Docker, Node.js(+npm), uv(또는 python3.11) 가 설치되어 있어야 함.
#
# ── 처음부터 실행하는 전체 순서 ───────────────────────────────────
#   docker compose up -d db redis                         # 1) DB·Redis
#   uv venv --python 3.11 venv311                         # 2) 파이썬 환경
#   uv pip install --python venv311 -r requirements-service.txt
#   uv pip install --python venv311 \
#       --index-url https://download.pytorch.org/whl/cpu torch==2.12.0 torchvision==0.27.0
#   cd backend && ../venv311/bin/python -m uvicorn app.main:app --port 8000   # 3) 웹
#   cd backend && ../venv311/bin/celery -A app.worker worker --concurrency=1  # 4) 워커
#   cd ui && npm install && npm run dev                   # 5) 프론트엔드
#
# torch/torchvision 은 PyTorch 전용 인덱스에서 받아야 하므로 "별도 스텝"으로 설치합니다.
# (PyPI 일반 패키지와 인덱스가 달라 한 파일에 섞으면 uv 인덱스 정책과 충돌)
#
# ── 설치 순서 (uv 권장) ───────────────────────────────────────
#   # 1) Python 3.11 가상환경
#   uv venv --python 3.11 venv311
#   # 2) 일반 의존성 (이 파일)
#   uv pip install --python venv311 -r requirements-service.txt
#   # 3) torch/torchvision (CPU 빌드)
#   uv pip install --python venv311 \
#       --index-url https://download.pytorch.org/whl/cpu torch==2.12.0 torchvision==0.27.0
#
# ── 일반 venv/pip 사용 시 ─────────────────────────────────────
#   python3.11 -m venv venv311 && source venv311/bin/activate   # (Win: venv311\Scripts\activate)
#   pip install -r requirements-service.txt
#   pip install --index-url https://download.pytorch.org/whl/cpu torch==2.12.0 torchvision==0.27.0
#
# ※ macOS: 위 torch 명령 그대로 사용 가능(맥은 CPU 휠이 기본).
# ※ GPU(CUDA)로 추론하려면 --index-url 을 https://download.pytorch.org/whl/cu121 등으로 바꾸고
#    model/config.py 의 INFER_DEVICE_TYPE="cuda" 로 설정.
# ================================================================
